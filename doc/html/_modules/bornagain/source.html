        self.pulse_energy = 0
        self.n_photons = 0
        self.profile = "tophat"
        self.mean_fluence = 0

    def __str__(self):

        s = ""
        if self.wavelength is not None:
            s += "wavelength : %g\n" % self.wavelength
        else:
            s += "wavelength : None\n"
        s += "B : [%g, %g, %g]\n" % (self.B[0], self.B[1], self.B[2])
        return s
